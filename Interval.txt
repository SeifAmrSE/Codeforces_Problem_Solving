#include <iostream>
using namespace std;

int main()
{
  double x;
  cin>>x;

  if(x >= 0 && x <= 25.00)
    cout<<"Interval [0,25]";

  else if(x > 25.00 && x <= 50.00)
    cout<<"Interval (25,50]";

  else if(x >= 50.00 && x <= 75.00)
    cout<<"Interval (50,75]";

  else if (x >= 75.00 && x <= 100.00)
    cout<<"Interval (75,100]";

  else if (x > 100.00)
        cout<<"Out of Interval";

  else
      cout<<"Out of Interval";
}
