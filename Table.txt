#include <iostream>
using namespace std;

int main()
{
	int N;
	cin >> N;

	if (N == 61)
         {
		cout << "Art of computer programming \n";
         }
	else if (N == 71)
         {
		cout << "Introduction to algorithm \n";
         }
	else if (N == 11)
         {
		cout << "Competitive programming \n";
         }
	else if (N == 29)
         {
		cout << "Programming challenges \n";
         }
	else
         {
		cout << "Not found \n";
         }
	return 0;
}
